SECTION 03

Data Ingestion Pipeline

This is the most underestimated step. Bad chunking = bad retrieval = bad answers, regardless of how good your LLM is.

◈ Ingestion Pipeline — Raw Data → Vector Store
RAW DOCS (./data/ at project root) ───────────────────────────────────────────────────── 📄 report.pdf ─┐ 📝 faq.txt ─┤ 📊 data.csv ─┤→ [ LOADER ] → [ CHUNKER ] → [ EMBEDDER ] → [ STORE ] 📋 records.xlsx ─┤ 📦 items.json ─┘ WHAT EACH FORMAT PRODUCES: ══════════════════════════ PDF → one Document per page TXT → one Document for the whole file CSV → one Document per row XLSX → one Document per row (pandas: "Column: Value Column2: Value2 ...") JSON → list → one Document per item object → one Document for whole file CHUNKER (the critical step): ════════════════════════════ ❌ BAD — too large (entire page): ┌────────────────────────────────────────────────────────────────┐ │ "Chapter 1: Ibuprofen for pain. Chapter 2: Side Effects... │ │ Chapter 3: Dosage... Chapter 4: Interactions..." │ └────────────────────────────────────────────────────────────────┘ → LLM gets too much noise, "lost in the middle" kicks in ✅ GOOD — focused chunks (~500 chars, with overlap): ┌──────────────────────────────────────────┐ │ Chunk 1: "Ibuprofen used for pain relief │ │ and fever reduction..." │ └──────────────────────────────────────────┘ ┌──────────────────────────────────────────┐ ← 50 char overlap │ Chunk 2: "...fever reduction. Side │ (prevents split context) │ effects include nausea, dizziness..." │ └──────────────────────────────────────────┘ Each chunk → 1 embedding vector stored in DB
backend/ingest.py 
import json
import os

import pandas as pd
from config import settings
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_community.document_loaders import (
    CSVLoader, PyPDFLoader, TextLoader,
)
from langchain_core.documents import Document


def load_documents(data_dir: str) -> list:
    """Load PDFs, text, CSV, Excel, and JSON files from a directory."""
    documents = []

    for filename in os.listdir(data_dir):
        filepath = os.path.join(data_dir, filename)

        if filename.endswith(".pdf"):
            loader = PyPDFLoader(filepath)
            documents.extend(loader.load())

        elif filename.endswith(".txt"):
            loader = TextLoader(filepath)
            documents.extend(loader.load())

        elif filename.endswith(".csv"):
            loader = CSVLoader(filepath)
            documents.extend(loader.load())

        elif filename.endswith(".xlsx") or filename.endswith(".xls"):
            # pandas read_excel → each row becomes one Document
            # page_content format: "Column1: val1 Column2: val2 ..."
            df = pd.read_excel(filepath)
            for index, row in df.iterrows():
                content = " ".join([f"{col}: {val}" for col, val in row.items()])
                doc = Document(
                    page_content=content,
                    metadata={"source": filepath, "row": index}
                )
                documents.append(doc)

        elif filename.endswith(".json"):
            with open(filepath, "r", encoding="utf-8") as f:
                json_data = json.load(f)

            if isinstance(json_data, list):
                # list → one Document per item
                for index, item in enumerate(json_data):
                    documents.append(Document(
                        page_content=json.dumps(item),
                        metadata={"source": filepath, "item": index},
                    ))
            else:
                # object → one Document for the whole file
                documents.append(Document(
                    page_content=json.dumps(json_data),
                    metadata={"source": filepath},
                ))

    print(f"Loaded {len(documents)} total documents/rows from {data_dir}")
    return documents


def chunk_documents(documents: list) -> list:
    """
    Split documents into chunks.

    WHY RecursiveCharacterTextSplitter?
    - Tries to split at natural boundaries: paragraph → sentence → word
    - Smarter than splitting at exactly 500 chars (avoids cutting mid-sentence)
    """
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=settings.chunk_size,         # 500 chars
        chunk_overlap=settings.chunk_overlap,   # 50 char overlap
        separators=["\n\n", "\n", ". ", " ", ""]  # priority order
    )

    chunks = splitter.split_documents(documents)
    print(f"Created {len(chunks)} chunks")
    return chunks


# ── Smoke test ──────────────────────────────────────────────────
# Run: export PYTHONPATH=$PYTHONPATH:$(pwd)/backend
# Then: python backend/ingest.py
if __name__ == "__main__":
    docs = load_documents("./data")          # data/ at project root

    print(f"First chunk:\n{chunks[0].page_content}")
⚡

Why overlap matters: If a sentence spans two chunks without overlap, both chunks are incomplete — the retriever can't find the full answer. 50–100 char overlap prevents this at minimal cost.

⚡

Excel metadata keys: Each Excel row gets metadata={"source": filepath, "row": index}. Each JSON list item gets {"source": filepath, "item": index}. Keep these keys consistent — you'll use them for metadata filtering later.

from langchain_community.embeddings import HuggingFaceEmbeddings
from langchain_google_genai import GoogleGenerativeAIEmbeddings
from config import settings


def get_embedder(use_google: bool = False):
    """
    Returns the embedding model.

    use_google=False (default):
        Runs locally via HuggingFace — completely free, no API key.
        Downloads BAAI/bge-small-en-v1.5 on first run (~130MB).

    use_google=True:
        Uses Google's embedding API — free tier, needs GOOGLE_API_KEY.
        Slightly higher quality but requires internet on every call.
    """
    if use_google:
        # Google Generative AI Embeddings — free tier
        # Requires: pip install langchain-google-genai
        return GoogleGenerativeAIEmbeddings(
            model="models/embedding-001",
            google_api_key=settings.google_api_key
        )
    else:
        # Local HuggingFace model — fully free, works offline
        # Best balance of quality + speed for most projects
        return HuggingFaceEmbeddings(
            model_name="BAAI/bge-small-en-v1.5",
            model_kwargs={"device": "cpu"},   # use "cuda" if you have a GPU
            encode_kwargs={"normalize_embeddings": True}  # improves cosine sim scores
        )


# ── Manual embedding example (to understand what's happening) ──
if __name__ == "__main__":
    embedder = get_embedder(use_google=False)  # local, free

    text = "What are the side effects of ibuprofen?"
    vector = embedder.embed_query(text)

    print(f"Text: {text}")
    print(f"Vector dims: {len(vector)}")         # → 384
    print(f"First 5 values: {vector[:5]}")        # → [0.23, -0.11, ...]
    print(f"Type: {type(vector)}")               # → list[float]
⚠️

Critical rule: Always use the same embedding model for ingestion AND queries. If you ingest docs with HuggingFace and query with Google embeddings, the vector spaces are incompatible — you'll get garbage results.

backend/vector_store.py — 🔲 next file to implement 
from langchain_community.vectorstores import Chroma
from embedder import get_embedder
from config import settings


def get_vector_store():
    """
    Returns a Chroma vector store instance.
    Creates it if it doesn't exist, loads it if it does.
    """
    embedder = get_embedder(use_google=False)  # local HuggingFace, free

    vectorstore = Chroma(
        collection_name=settings.collection_name,
        embedding_function=embedder,
        persist_directory=settings.chroma_persist_dir  # saves to disk
    )
    return vectorstore


def ingest_to_store(chunks: list) -> None:
    """
    Takes document chunks and stores them in the vector DB.
    Run this once (or when you add new documents).
    """
    embedder = get_embedder(use_google=False)  # must match get_vector_store()

    # This automatically:
    # 1. Generates embeddings for each chunk
    # 2. Stores embedding + raw text + metadata
    # 3. Builds the HNSW index
    vectorstore = Chroma.from_documents(
        documents=chunks,
        embedding=embedder,
        collection_name=settings.collection_name,
        persist_directory=settings.chroma_persist_dir
    )

    print(f"✅ Stored {len(chunks)} chunks in ChromaDB")
    return vectorstore

SECTION 06

Query Pipeline

Every time a user sends a message, this pipeline runs. Embed the query → find nearest vectors → return relevant chunks.

◈ Query Flow — User Message → Retrieved Context
USER: "What happens if I take ibuprofen on empty stomach?" │ ▼ ┌─────────────────────┐ │ EMBED QUERY │ → [0.41, -0.08, 0.75, ...] (same model as ingestion!) └─────────────────────┘ │ ▼ ┌─────────────────────────────────────────────────────────┐ │ SIMILARITY SEARCH in ChromaDB │ │ │ │ Compute cosine similarity between query vector │ │ and ALL stored vectors → return top-K closest │ │ │ │ Score: 0.96 "Avoid on empty stomach. Side effects..."│ ← returned │ Score: 0.91 "Ibuprofen recommended 500mg for pain..."│ ← returned │ Score: 0.83 "Do not mix ibuprofen with alcohol..." │ ← returned │ Score: 0.41 "Paris is a beautiful city..." │ ← NOT returned └─────────────────────────────────────────────────────────┘ │ ▼ TOP-K CHUNKS (the "context") passed to LLM

from backend.vector_store import get_vector_store
from backend.config import settings


def retrieve(query: str, top_k: int = settings.top_k) -> list:
    """
    Core retrieval function.
    Returns top-k most relevant document chunks for a query.
    """
    vectorstore = get_vector_store()

    # similarity_search_with_score also returns the similarity score
    results = vectorstore.similarity_search_with_score(
        query=query,
        k=top_k
    )

    return results  # list of (Document, float) tuples


if __name__ == "__main__":
    query = "What are the side effects of ibuprofen?"
    results = retrieve(query)

    print(f"\n🔍 Query: {query}\n")
    print("=" * 60)

    for i, (doc, score) in enumerate(results):
        print(f"\n[Rank {i+1}] Similarity: {score:.3f}")
        print(f"Source: {doc.metadata.get('source', 'unknown')}")
        print(f"Content: {doc.page_content[:200]}...")
        print("-" * 40)

SECTION 07

Full RAG Implementation

Now we connect retrieval to the LLM. This is the complete end-to-end pipeline your chatbot will run on every message.

◈ Complete RAG Flow
USER MESSAGE │ ▼ [1] EMBED QUERY → query_vector │ ▼ [2] VECTOR SEARCH → top-4 relevant chunks │ ▼ [3] BUILD PROMPT: ┌────────────────────────────────────────────────────┐ │ SYSTEM: You are a helpful assistant. Answer ONLY │ │ using the provided context. If not found, │ │ say "I don't have that information." │ │ │ │ CONTEXT: │ │ [Chunk 1: "Avoid on empty stomach..."] │ │ [Chunk 2: "Side effects: nausea, dizziness..."] │ │ [Chunk 3: "Do not mix with alcohol..."] │ │ │ │ USER: "What happens if I take ibuprofen on │ │ empty stomach?" │ └────────────────────────────────────────────────────┘ │ ▼ [4] LLM (Gemini 1.5 Flash — free tier) │ ▼ [5] GROUNDED RESPONSE → back to user
backend/rag_engine.py — 🔲 planned (implement after vector_store.py)
from langchain_google_genai import ChatGoogleGenerativeAI from langchain.prompts import ChatPromptTemplate from langchain.schema.runnable import RunnablePassthrough from langchain.schema.output_parser import StrOutputParser from vector_store import get_vector_store from config import settings # ── 1. THE PROMPT TEMPLATE ────────────────────────────────────── # This is where most developers get lazy. A good prompt makes # the difference between a hallucinating bot and an accurate one. SYSTEM_PROMPT = """You are a knowledgeable and helpful AI assistant. Your rules: 1. Answer ONLY using the provided context below. 2. If the answer is not in the context, say: "I don't have enough information to answer that accurately." 3. Be concise. Cite context when possible. 4. Never make up information. Context: {context}""" prompt = ChatPromptTemplate.from_messages([ ("system", SYSTEM_PROMPT), ("human", "{question}") ]) # ── 2. FORMAT RETRIEVED DOCS ──────────────────────────────────── def format_docs(docs) -> str: """Join retrieved chunks into a single context string.""" return "\n\n---\n\n".join([doc.page_content for doc in docs]) # ── 3. BUILD THE RAG CHAIN ────────────────────────────────────── def build_rag_chain(): """ Creates the full RAG chain using LangChain's LCEL syntax. Chain: query → retriever → format → prompt → LLM → parse """ vectorstore = get_vector_store() # Retriever: wraps vector store for easy use in chains retriever = vectorstore.as_retriever( search_type="similarity", search_kwargs={"k": settings.top_k} ) # LLM: Gemini 1.5 Flash — free tier, fast, handles long contexts well # temperature=0.1 → more factual, less creative llm = ChatGoogleGenerativeAI( model=settings.model_name, # "gemini-1.5-flash" temperature=settings.temperature, google_api_key=settings.google_api_key, convert_system_message_to_human=True # required for Gemini ) # LCEL Chain: connects each step with pipe operator | rag_chain = ( { "context": retriever | format_docs, # retrieves + formats "question": RunnablePassthrough() # passes query through } | prompt # fills the template | llm # sends to LLM | StrOutputParser() # extracts text from response ) return rag_chain # ── 4. MAIN ASK FUNCTION ──────────────────────────────────────── def ask(question: str) -> str: """Single entry point for the RAG chatbot.""" chain = build_rag_chain() return chain.invoke(question) # ── Quick test ───────────────────────────────────────────────── if __name__ == "__main__": answer = ask("What are the side effects of ibuprofen?") print(f"\n🤖 Answer:\n{answer}")
⚡

About the prompt: The instruction "Answer ONLY using the provided context" is critical. Without it, the LLM will blend your retrieved data with its training data and hallucinate. Force it to be grounded.

SECTION 09

Production-Level Improvements

The basic RAG works. Now here's what separates a toy from a real product.

01

Caching Repeated Queries

If 100 users ask "what are your pricing plans?", you pay for 100 LLM calls. Cache responses with Redis or even a simple Python dict. TTL = 1 hour for stable content.

02

Hybrid Search (Vector + Keyword)

Pure vector search misses exact matches (product codes, names). Combine BM25 keyword search + vector similarity. Weaviate does this natively. For Chroma, use LangChain's EnsembleRetriever.

03

Metadata Filtering

Don't retrieve from the whole DB. Filter first: where={"category": "medical"} then search. Dramatically improves accuracy and speed. Think of it as SQL WHERE + vector similarity combined.

04

Streaming Responses

Instead of waiting 3s for the full response, stream tokens as they generate. Use chain.stream(question) + FastAPI's StreamingResponse. Users perceive this as much faster.

05

Chat History (Conversational RAG)

Add conversation memory. Store last N messages in Redis, inject them into the prompt. Use LangChain's ConversationalRetrievalChain for this pattern.

Metadata filtering example
# Filter by category BEFORE doing similarity search # Much more accurate for multi-domain systems like yours retriever = vectorstore.as_retriever( search_type="similarity", search_kwargs={ "k": 4, "filter": {"category": "medical"} # only search medical docs } ) # For tourism panel: # "filter": {"category": "tourism"} # For farmer panel: # "filter": {"category": "agriculture"}
Streaming response in FastAPI

SECTION 10

Common Mistakes to Avoid

These mistakes account for 80% of RAG system failures. Learn them now, not after 3 weeks of debugging.

❌ MISTAKE 01

Chunks Too Large or Too Small

Huge chunks = noisy context, "lost in the middle". Tiny chunks (50 chars) = no context, LLM can't answer. Sweet spot: 400-600 chars with 50-100 overlap.

✅ Fix: Use RecursiveCharacterTextSplitter, test with 500/50.
❌ MISTAKE 02

Different Embedding Models

Ingesting with Google embeddings and querying with HuggingFace. The vector spaces are completely different — results will be random garbage.

✅ Fix: One embedder, one config, used everywhere.
❌ MISTAKE 03

Not Testing Retrieval First

Adding the LLM before verifying the retriever returns relevant chunks. If retrieval is broken, adding Gemini on top doesn't help.

✅ Fix: Always print and review retrieved chunks before connecting the LLM.
❌ MISTAKE 04

No "I don't know" Fallback

Without explicit instructions, LLMs confidently hallucinate answers from training data when context doesn't have the answer.

✅ Fix: System prompt must say "answer ONLY from context, otherwise say you don't know."
❌ MISTAKE 05

Rebuilding the DB Every Request

Calling Chroma.from_documents() in the query path re-ingests everything on every user message. Catastrophically slow and expensive.

✅ Fix: Ingest ONCE. Query uses Chroma() constructor to load existing DB.
❌ MISTAKE 06

No Evaluation

Shipping without measuring accuracy. Your RAG can silently degrade as you add more documents if you never test it.

✅ Fix: Build a small test set (20 Q&A pairs). Run it after every ingestion change. Use RAGAS library for automated RAG evaluation.

SECTION 11

Chat Memory (Conversational RAG)

The basic RAG forgets every message the moment it answers. Real chatbots remember the conversation. Here's how to add two-tier memory — short-term for the current session and long-term retrieval for past context.

◈ Two-Tier Memory Architecture
SHORT-TERM MEMORY (current session) ───────────────────────────────────── Stored in: Python list / Redis Contains: Last 6–10 messages Injected: Into every prompt automatically Expires: When session ends LONG-TERM MEMORY (across sessions) ───────────────────────────────────── Stored in: ChromaDB (separate collection) Contains: Embedded past conversations Injected: Top-3 semantically relevant memories Expires: Never (until manually cleared) PROMPT STRUCTURE WITH MEMORY: ══════════════════════════════ ┌─────────────────────────────────────────────────────┐ │ SYSTEM: You are a helpful assistant. │ │ Answer using context + conversation below. │ │ │ │ CONTEXT (from RAG): │ │ [retrieved doc chunks...] │ │ │ │ RELEVANT PAST MEMORY: │ │ User previously asked about ibuprofen dosage. │ │ │ │ CONVERSATION HISTORY: │ │ Human: What are the side effects? │ │ AI: Nausea and dizziness are common. │ │ Human: Can I take it daily? ← current question │ └─────────────────────────────────────────────────────┘
backend/memory.py — short-term session memory 
from collections import defaultdict
from langchain.schema import HumanMessage, AIMessage

# In-memory store: { chat_id: [messages] }
# Replace with Redis for production multi-instance deployments
_sessions: dict = defaultdict(list)

MAX_HISTORY = 10  # keep last 10 messages per session


def get_history(chat_id: str) -> list:
    """Return conversation history for a session."""
    return _sessions[chat_id]


def add_turn(chat_id: str, question: str, answer: str) -> None:
    """Append a human/AI turn and trim to MAX_HISTORY."""
    _sessions[chat_id].append(HumanMessage(content=question))
    _sessions[chat_id].append(AIMessage(content=answer))

    # Keep only the last MAX_HISTORY messages
    if len(_sessions[chat_id]) > MAX_HISTORY:
        _sessions[chat_id] = _sessions[chat_id][-MAX_HISTORY:]


def clear_history(chat_id: str) -> None:
    """Wipe a session — useful for "start over" button."""
    _sessions.pop(chat_id, None)


def format_history_for_prompt(chat_id: str) -> str:
    """Convert history list into a readable string for the prompt."""
    history = get_history(chat_id)
    if not history:
        return "No previous conversation."

    lines = []
    for msg in history:
        role = "Human" if isinstance(msg, HumanMessage) else "AI"
        lines.append(f"{role}: {msg.content}")
    return "\n".join(lines)
backend/rag_engine.py — updated with memory support 
from langchain_google_genai import ChatGoogleGenerativeAI
from langchain.prompts import ChatPromptTemplate
from langchain.schema.runnable import RunnablePassthrough
from langchain.schema.output_parser import StrOutputParser
from vector_store import get_vector_store
from memory import get_history, add_turn, format_history_for_prompt
from config import settings


# ── Prompt now includes conversation history ─────────────────────
SYSTEM_PROMPT = """You are a knowledgeable and helpful AI assistant.

Your rules:
1. Answer using the provided context AND the conversation history.
2. If the answer is not in the context, say:
   "I don't have enough information to answer that accurately."
3. Use conversation history to understand follow-up questions.
4. Never make up information.

Context (from documents):
{context}

Conversation so far:
{history}"""

prompt = ChatPromptTemplate.from_messages([
    ("system", SYSTEM_PROMPT),
    ("human", "{question}")
])


def format_docs(docs) -> str:
    return "\n\n---\n\n".join([doc.page_content for doc in docs])


def ask(question: str, chat_id: str = "default") -> str:
    """
    Full RAG + memory pipeline.
    chat_id lets you track separate conversations per user.
    """
    vectorstore = get_vector_store()
    retriever   = vectorstore.as_retriever(search_kwargs={"k": settings.top_k})

    llm = ChatGoogleGenerativeAI(
        model=settings.model_name,
        temperature=settings.temperature,
        google_api_key=settings.google_api_key,
        convert_system_message_to_human=True
    )

    # Retrieve docs + fetch current session history
    docs    = retriever.invoke(question)
    context = format_docs(docs)
    history = format_history_for_prompt(chat_id)

    # Build and run the chain
    chain  = prompt | llm | StrOutputParser()
    answer = chain.invoke({
        "context":  context,
        "history":  history,
        "question": question
    })

    # Save this turn so the next question can reference it
    add_turn(chat_id, question, answer)

    return answer
backend/api.py — updated /ask endpoint with chat_id 
@app.post("/ask", response_model=AskResponse)
async def ask_endpoint(req: AskRequest):
    if not req.question.strip():
        raise HTTPException(status_code=400, detail="Question cannot be empty")
    try:
        start  = time.time()
        answer = ask(req.question, chat_id=req.chat_id)  # passes chat_id
        latency = (time.time() - start) * 1000
        return AskResponse(answer=answer, latency_ms=round(latency, 2), chat_id=req.chat_id)
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
